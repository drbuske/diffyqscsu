\begin{equation*} \bigl((\omega_0^2 - \omega^2)B - 2\omega p A\bigr) \sin (\omega t) + \bigl((\omega_0^2 - \omega^2)A + 2\omega p B\bigr) \cos (\omega t) = \frac{F_0}{m} \cos (\omega t) . \end{equation*}
